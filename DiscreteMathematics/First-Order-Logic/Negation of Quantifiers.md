$$
\neg \; \forall x\; \alpha \equiv \exists \neg \; \alpha
$$
$$
\exists \neg \; \alpha \equiv \neg \; \forall x\; \alpha
$$
$$
\neg \forall x\; \alpha \not\equiv \neg \exists x\; \alpha 
$$
